def window(x,y):
    t.penup()
    t.goto(x,y)
    t.pendown()
    t.fillcolor("#3c3c3c")
    t.begin_fill()
    t.right(90)
    t.fd(70)
    t.right(90)
    t.fd(30)
    t.right(90)
    t.fd(70)
    t.right(90)
    t.fd(30)
    t.end_fill()

window(-220,50)

def light(x,y):
    t.penup()
    t.goto(x, y)
    t.pd()
    t.fillcolor("#FAFA96")
    t.begin_fill()
    t.left(180)
    t.circle(12,180)
    t.end_fill()
    
light(-250,90)

def square(x):
    t.penup()
    t.goto(-400+x,160)
    t.pendown()
    t.fillcolor("#004442")
    t.begin_fill()
    t.fd(20)
    t.right(90)
    t.fd(40)
    t.right(90)
    t.fd(20)
    t.right(90)
    t.fd(40)
    t.end_fill()
    
def minsquare(x):
    t.penup()
    t.goto(-410+x,120)
    t.fd(20)
    t.left(90)
    t.fd(40)
    t.left(90)
    t.fd(20)
    t.left(90)
    t.fd(40)

    
for i in range(1, 53):
    n=i*15
    if i%2==1:
        square(n)
    else:
         minsquare(n)



#네온사인

def snow():
    randx=r.randint(-400,400)
    randy=r.randint(-100,380)
    t.pencolor(0,0,0)
    t.penup()
    t.goto(randx,randy)
    t.pendown()
    t.fillcolor("#F9FFFF")
    t.begin_fill()
    t.circle(5)
    t.end_fill()

def snowhigh():
    randx=r.randint(-400,400)
    randy=r.randint(-250,380)
    t.pencolor(0,0,0)
    t.penup()
    t.goto(randx,randy)
    t.pendown()
    t.fillcolor("#F9FFFF")
    t.begin_fill()
    t.circle(5)
    t.end_fill()

for i in range(20):
    snow()
for i in range(7):
    snowhigh()
